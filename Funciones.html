#Crea una función que muestre un mensaje de bienvenida personalizado.

#Crea una función que reciba un número y determine si es par o impar.
def es_par(numero):
    
    if numero % 2 == 0:
        return "Par"
    else:
        return "Impar"

num = int(input("Ingresa un número: "))
print(es_par(num))

#Crea una función que reciba dos números y retorne su suma
def suma(a, b):
    return a + b 
    
a = float(input("Ingresa el primer número: "))
b = float(input("Ingresa el segundo número: "))
print("Resultado: ", suma(a, b))

#Crea una función que reciba una palabra y retorne su longitud
def longitud_palabra(palabra):
    return len(palabra)
    
texto = input("Ingresar una palabra: ")
print("Longitud: ", longitud_palabra(texto))

#Crea una función que reciba un número y retorne su cuadrado

    return numero ** 2  
    
num = float(input("Ingresa un número: "))
print("Cuadrado: ", cuadrado(num))

#Crea una función que reciba tres números y retorne el mayor de ellos
def mayor_de_tres(a, b, c):
    return max(a, b, c)

a = float(input("Número 1: "))
b = float(input("Número 2: "))
c = float(input("Número 3: "))
print("Mayor: ", mayor_de_tres(a, b, c))

#Crea una función que convierta una temperatura de Celsius a Fahrenheit
def celsius_a_fahrenheit(c):
    return (c * 9/5) + 32

temp = float(input("Ingrese la temperatura en Celsius: "))
print("Temperatura en Fahrenheit: ", celsius_a_fahrenheit(temp))

#Crea una función que reciba una lista de números y retorne la suma total
def suma_lista(lista):
    total = 0
    for num in lista:
        total += num
    return total
    
numeros = []
for i in range(8):
    num = float(input(f"Ingrese un número: "))
    numeros.append(num)
print("Suma total: ", suma_lista(numeros))

#Crea una función que retorne True si un número es positivo, o False si no lo es
def es_positivo(numero):
    return numero > 0
    
num = float(input("Ingrese un número: "))

if es_positivo(num):
    print("True")
else:
    print("False")

#Crea una función que reciba una palabra y retorne la cadena invertida
def invertir_palabra(palabra):
    return palabra[::-1]
    
palabra = input("Ingrese una palabra: ")
print("Palabra invertida: ", invertir_palabra(palabra))
